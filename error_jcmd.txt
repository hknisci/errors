#!/usr/bin/env bash
set -euo pipefail

# =========================
# Config
# =========================
OUT_DIR="${OUT_DIR:-./jvm_flags_dump}"
TS="$(date +%Y%m%d_%H%M%S)"
VM_FILE="${VM_FILE:-$OUT_DIR/jcmd_vmflags_${TS}.tsv}"
GC_FILE="${GC_FILE:-$OUT_DIR/jcmd_gcflags_${TS}.tsv}"

# Parallelism
JOBS="${JOBS:-10}"

# oc request timeout (best effort)
OC_TIMEOUT="${OC_TIMEOUT:-20s}"

# If 1: do not exec into pods, only list targets
DRY_RUN="${DRY_RUN:-0}"

mkdir -p "$OUT_DIR"

log() { echo "[$(date +%H:%M:%S)] $*" >&2; }

need() { command -v "$1" >/dev/null 2>&1 || { echo "ERROR: missing: $1" >&2; exit 1; }; }
need oc; need awk; need sed; need sort; need head; need tr; need xargs; need file

if file "$0" 2>/dev/null | grep -qi 'CRLF'; then
  log "WARNING: CRLF detected. Fix with: sed -i 's/\r$//' $0"
fi

# Same headers for BOTH files so they match 1:1
HEADER="namespace\tdc\tpod\tcontainer\tstatus\tjava_version\tjcmd_vm_version\tvm_flags\tgc_related\tnote"
printf "%b\n" "$HEADER" > "$VM_FILE"
printf "%b\n" "$HEADER" > "$GC_FILE"

oc_safe() {
  oc --request-timeout="$OC_TIMEOUT" "$@" 2>&1
}

# Convert multi-line output to single line (keeps content, replaces newlines with " | ")
one_line() {
  tr -d '\r' | tr '\t' ' ' | awk '{printf "%s%s", (NR==1?"":" | "), $0}'
}

# Extract GC-related hints from flags (best effort)
extract_gc() {
  tr ' ' '\n' | \
    grep -E 'Use(G1|Z|Shenandoah|Parallel|ConcMarkSweep)GC|MaxGCPauseMillis|InitiatingHeapOccupancyPercent|G1(H|E|R)|ZGC|Shenandoah|Xms|Xmx' \
    2>/dev/null | head -n 200 | awk '{printf "%s%s", (NR==1?"":" "), $0}'
}

write_row_both() {
  local ns="$1" dc="$2" pod="$3" ctr="$4" status="$5" jver="$6" vmver="$7" flags="$8" gc="$9" note="${10}"

  printf "%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\n" \
    "$ns" "$dc" "$pod" "$ctr" "$status" "$jver" "$vmver" "$flags" "$gc" "$note" >> "$VM_FILE"

  # Exactly same rows in GC file (no mismatch)
  printf "%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\n" \
    "$ns" "$dc" "$pod" "$ctr" "$status" "$jver" "$vmver" "$flags" "$gc" "$note" >> "$GC_FILE"
}

collect_one_dc() {
  local ns="$1" dc="$2"

  if [ "$DRY_RUN" = "1" ]; then
    write_row_both "$ns" "$dc" "-" "-" "DRY_RUN" "-" "-" "-" "-" "dry_run_no_exec"
    return 0
  fi

  # Find latest revision -> RC name: <dc>-<rev>
  local rev rc pod ctr
  rev="$(oc_safe -n "$ns" get dc "$dc" -o jsonpath='{.status.latestVersion}' 2>/dev/null | tr -d '\r' || true)"
  if [ -z "${rev:-}" ]; then
    write_row_both "$ns" "$dc" "-" "-" "NO_REVISION" "-" "-" "-" "-" "dc_has_no_status_latestVersion_or_forbidden"
    return 0
  fi
  rc="${dc}-${rev}"

  # Pick first Running pod owned by that RC (ownerReferences[0].name == rc)
  pod="$(oc_safe -n "$ns" get pods --field-selector=status.phase=Running \
      -o jsonpath="{range .items[?(@.metadata.ownerReferences[0].name=='$rc')]}{.metadata.name}{'\n'}{end}" \
      2>/dev/null | tr -d '\r' | head -n1 || true)"

  if [ -z "${pod:-}" ]; then
    write_row_both "$ns" "$dc" "-" "-" "POD_NOT_FOUND" "-" "-" "-" "-" "no_running_pod_found_for_rc"
    return 0
  fi

  ctr="$(oc_safe -n "$ns" get pod "$pod" -o jsonpath='{.spec.containers[0].name}' 2>/dev/null | tr -d '\r' || true)"
  [ -z "${ctr:-}" ] && ctr="-"

  # RBAC check for pod
  local chk
  chk="$(oc_safe -n "$ns" get pod "$pod" -o name || true)"
  if ! echo "$chk" | grep -q "^pod/"; then
    if echo "$chk" | grep -qi "forbidden"; then
      write_row_both "$ns" "$dc" "$pod" "$ctr" "FORBIDDEN" "-" "-" "-" "-" "get_pod_forbidden"
    else
      write_row_both "$ns" "$dc" "$pod" "$ctr" "POD_GET_ERROR" "-" "-" "-" "-" "$(echo "$chk" | one_line | head -c 400)"
    fi
    return 0
  fi

  # java -version (even if jcmd missing)
  local java_ver java_check
  java_ver="$(oc_safe -n "$ns" exec "$pod" -c "$ctr" -- sh -lc 'java -version 2>&1' 2>/dev/null | one_line || true)"
  if [ -z "${java_ver:-}" ]; then
    java_check="$(oc_safe -n "$ns" exec "$pod" -c "$ctr" -- sh -lc 'command -v java >/dev/null 2>&1; echo $?' || true)"
    if echo "$java_check" | grep -qi "forbidden"; then
      write_row_both "$ns" "$dc" "$pod" "$ctr" "FORBIDDEN" "-" "-" "-" "-" "exec_forbidden_on_java_check"
      return 0
    fi
    if echo "$java_check" | tail -n 1 | grep -q "^0$"; then
      java_ver="java_present_but_version_empty"
    else
      java_ver="NO_JAVA"
    fi
  fi

  # jcmd presence check (capture exact command and output for NO_JCMD explanation)
  local jcmd_check_cmd jcmd_check_out jcmd_rc
  jcmd_check_cmd="sh -lc 'command -v jcmd >/dev/null 2>&1; echo \$?'"
  jcmd_check_out="$(oc_safe -n "$ns" exec "$pod" -c "$ctr" -- sh -lc 'command -v jcmd >/dev/null 2>&1; echo $?' || true)"
  jcmd_rc="$(echo "$jcmd_check_out" | tail -n 1 | tr -d '\r' | tr -d '\n')"

  if echo "$jcmd_check_out" | grep -qi "forbidden"; then
    write_row_both "$ns" "$dc" "$pod" "$ctr" "FORBIDDEN" "$java_ver" "-" "-" "-" "exec_forbidden_on_jcmd_check; cmd=$jcmd_check_cmd"
    return 0
  fi

  if [ "${jcmd_rc:-1}" != "0" ]; then
    write_row_both "$ns" "$dc" "$pod" "$ctr" "NO_JCMD" "$java_ver" "-" "-" "-" \
      "no_jcmd; executed=$jcmd_check_cmd; result=$(echo "$jcmd_check_out" | one_line | head -c 300)"
    return 0
  fi

  # Determine Java PID (prefer PID 1 if it is java)
  local pid
  pid="$(oc_safe -n "$ns" exec "$pod" -c "$ctr" -- sh -lc '
    if [ -r /proc/1/cmdline ] && tr "\0" " " < /proc/1/cmdline | grep -qiE "(^| )java( |$)"; then
      echo 1; exit 0
    fi
    if command -v ps >/dev/null 2>&1; then
      p="$(ps -eo pid,args 2>/dev/null | grep -i java | grep -v grep | head -n1 | awk "{print \$1}" 2>/dev/null || true)"
      echo "$p" | grep -E "^[0-9]+$" >/dev/null 2>&1 && { echo "$p"; exit 0; }
    fi
    echo ""
  ' 2>/dev/null || true)"
  pid="$(echo "$pid" | tail -n 1 | tr -d '\r' | tr -d '\n')"
  [ -z "${pid:-}" ] && pid="1"

  # Best-effort ensure /tmp exists for attach socket (if FS is RO it will fail silently)
  oc_safe -n "$ns" exec "$pod" -c "$ctr" -- sh -lc 'mkdir -p /tmp 2>/dev/null || true' >/dev/null 2>&1 || true

  # jcmd VM.version
  local vmver_raw vmver
  vmver_raw="$(oc_safe -n "$ns" exec "$pod" -c "$ctr" -- sh -lc "TMPDIR=/tmp jcmd $pid VM.version 2>&1" || true)"
  vmver="$(printf "%s" "$vmver_raw" | one_line)"

  # jcmd VM.flags
  local flags_raw flags
  flags_raw="$(oc_safe -n "$ns" exec "$pod" -c "$ctr" -- sh -lc "TMPDIR=/tmp jcmd $pid VM.flags 2>&1" || true)"
  flags="$(printf "%s" "$flags_raw" | one_line)"

  # Status handling (includes your screenshot error)
  local status note
  status="OK"
  note="pid=$pid"

  if echo "$flags_raw" | grep -qi "AttachNotSupportedException"; then
    status="JCMD_ATTACH_NOT_SUPPORTED"
    note="AttachNotSupportedException; possible causes: non-HotSpot JVM, attach disabled (-XX:+DisableAttachMechanism), security context, different user, /tmp issues; pid=$pid"
  elif echo "$flags_raw" | grep -qiE "Unable to open socket file|doesn't respond within 10500ms"; then
    status="JCMD_ATTACH_TIMEOUT"
    note="attach_socket_timeout_or_unreachable; pid=$pid"
  elif echo "$flags_raw" | grep -qiE "Permission denied"; then
    status="JCMD_PERMISSION_DENIED"
    note="permission_denied_attach; pid=$pid"
  elif echo "$flags_raw" | grep -qiE "No such process"; then
    status="JCMD_NO_SUCH_PROCESS"
    note="pid_not_found; pid=$pid"
  elif echo "$flags_raw" | grep -qiE "forbidden"; then
    status="FORBIDDEN"
    note="exec_forbidden_on_jcmd"
  fi

  # GC extraction (always produce a row, even if empty)
  local gc
  gc="$(printf "%s" "$flags" | extract_gc || true)"
  [ -z "${gc:-}" ] && gc="-"

  # If flags output is empty for some reason, keep placeholder
  [ -z "${flags:-}" ] && flags="-"
  [ -z "${vmver:-}" ] && vmver="-"
  [ -z "${java_ver:-}" ] && java_ver="-"

  write_row_both "$ns" "$dc" "$pod" "$ctr" "$status" "$java_ver" "$vmver" "$flags" "$gc" "$note"
}

export -f collect_one_dc
export -f oc_safe
export -f one_line
export -f extract_gc
export -f write_row_both
export VM_FILE GC_FILE DRY_RUN OC_TIMEOUT

# =========================
# Build job list (ns, dc)
# =========================
log "Discovering namespaces..."
namespaces="$(oc_safe get ns -o jsonpath='{range .items[*]}{.metadata.name}{"\n"}{end}' | tr -d '\r' || true)"
if [ -z "${namespaces:-}" ]; then
  log "ERROR: cannot list namespaces (RBAC/oc issue)."
  exit 1
fi

tmp_jobs="$OUT_DIR/jobs_${TS}.txt"
: > "$tmp_jobs"

log "Discovering DeploymentConfigs..."
while IFS= read -r ns; do
  [ -z "$ns" ] && continue
  dcs="$(oc_safe -n "$ns" get dc -o jsonpath='{range .items[*]}{.metadata.name}{"\n"}{end}' 2>/dev/null | tr -d '\r' || true)"
  [ -z "${dcs:-}" ] && continue
  while IFS= read -r dc; do
    [ -z "$dc" ] && continue
    printf "%q %q\n" "$ns" "$dc" >> "$tmp_jobs"
  done <<< "$dcs"
done <<< "$namespaces"

if [ ! -s "$tmp_jobs" ]; then
  log "No DCs found or RBAC prevents listing."
  log "VM: $VM_FILE"
  log "GC: $GC_FILE"
  exit 0
fi

# =========================
# Run in parallel
# =========================
log "Running jobs in parallel: JOBS=$JOBS"
log "VM: $VM_FILE"
log "GC: $GC_FILE"

cat "$tmp_jobs" | xargs -P "$JOBS" -n 2 bash -lc 'collect_one_dc "$0" "$1"'

log "Done."
log "VM: $VM_FILE"
log "GC: $GC_FILE"
